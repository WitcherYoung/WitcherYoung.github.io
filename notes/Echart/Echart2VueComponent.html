    Echarts Vue组件化

    vuePress-theme-reco 夜慕    2019 - 2022

    Echarts Vue组件化

    夜慕 2021/12/28 echarts

    # 前言

    如今随着信息化的深入，产生的数据越来越多，分析数据可从已有数据中发现跟多问题。而分析挖掘数据便少不了图表。今天结合Vue和echarts，分享一下我是如何封装一个Echarts的Vue组件的。

    本组件基于 Vue 2.x

    相关链接
    Vue 2.x (opens new window)
    Echarts配置项手册 (opens new window)
    Echarts API (opens new window)

    # 设计思路

    首先简要梳理一下 echarts 图表的使用过程。将 配置对象 传给 Echarts，Echarts 渲染配置项对应图表，渲染后可能需要对图表实例以执行某些 实例方法 来完成特定需求。

    组件化便是要将业务解耦，因此需要将 配置项 作为 参数 传递给组件，可以将该组件划分成 初始化 接收参数 渲染 获取实例 4大功能块

    1. 初始化：实例容器，一般是一个具有高宽的 DIV 元素（也支持直接使用canvas元素作为容器）。在 Vue mounted 生命周期初始化 该DIV元素
    2. 接收数据：通过 watch 监听 接收参数 变化，从而触发图表渲染改变
    3. 渲染：通过 echarts 实例方法 setOption 来改变配置项，进而渲染图表
    4. 图表实例方法：每次在图表渲染完成之后，向父组件抛出事件 并提供 图表实例对象，使得父组件可直接使用 图表实例 的方法

const DEFAULT_CONFIG = {
  text: '加载中...',
  color: '#9fe6b8',
  textColor: '#FFF',
  maskColor: 'rgba(0, 0, 0, 0.8)',
  zlevel: 0,
  // 字体大小。从 `v4.8.0` 开始支持。
  fontSize: 14,
  // 是否显示旋转动画（spinner）。从 `v4.8.0` 开始支持。
  showSpinner: true,
  // 旋转动画（spinner）的半径。从 `v4.8.0` 开始支持。
  spinnerRadius: 10,
  // 旋转动画（spinner）的线宽。从 `v4.8.0` 开始支持。
  lineWidth: 5,
  // 字体粗细。从 `v5.0.1` 开始支持。
  fontWeight: 'normal',
  // 字体风格。从 `v5.0.1` 开始支持。
  fontStyle: 'normal',
  // 字体系列。从 `v5.0.1` 开始支持。
  fontFamily: 'sans-serif'
};

/** 显示图表loading
 * @params config 图表loading配置选项
 */
function showChartLoading(chart, config = {}) {
  if (!chart) {
    return;
  }
  return chart.showLoading({
    ...DEFAULT_CONFIG,
    ...config,
  });
}

/** 关闭图表loading
 * @params config 图表loading配置选项
 * @params delay  延时关闭时间
 */
function closeChartLoading(chart, delay = 500) {
  if (!chart) {
    return;
  }
  return setTimeout(() => {
    chart.hideLoading();
  }, delay);
}


export default {
  name: "echart",
  props: {
    className: {  // 样式类
      type: String,
      default: "chart",
    },
    id: { // 图表元素id
      type: String,
      default: "chart",
    },
    width: {  // 图表长度
      type: String,
      default: "300px",
    },
    height: { // 图表高度
      type: String,
      default: "300px",
    },
    options: {  // echarts option配置 详细参照 https://echarts.apache.org/zh/option.html#title
      type: Object,
      default: () => ({}),
    },
    loadingDelay: { // loading效果关闭延时
      type: Number,
      default: 500,
    },
    clear: {  // 是否清空当前实例（会移除实例中所有的组件和图表）
      type: Boolean,
      default: false,
    },
    showLoading: { // 是否展示loading效果
      type: Boolean,
      default: true,
    },
  },
  data() {
    return {
      chart: null,
    };
  },
  watch: {
    options: {
      handler(options) {
        this.$nextTick(() => this.setOption(options));
      },
      immediate: true,
      deep: true,
    },
  },
  mounted() {
    this.initChart();
  },
  methods: {
    initChart() {
      // 初始化echart
      this.chart = this.$echarts.init(this.$el);
      this.setOption(this.options);
    },
    setOption(options = {}) {
      this.showLoading && showChartLoading(this.chart);
      this.clear && this.chart.clear(); // 清空当前实例（会移除实例中所有的组件和图表）
      // 设置true清空echart缓存
      this.chart.setOption(options, true);
      setTimeout(() => {
        this.showLoading && closeChartLoading(this.chart);
        this.$emit("rendered", this.chart); // 向父组件抛出echart实例对象
      }, this.loadingDelay);
    },
  },
};
