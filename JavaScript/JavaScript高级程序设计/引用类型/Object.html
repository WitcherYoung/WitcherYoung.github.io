# 对象常用的方式

// 创建对象
var obj1 = new Object();
var obj2 = {
	name: "name",
	age: 12
};
// 访问对象属性
/* .表示法访问属性 */
obj2.name			
/* []法访问属性, []中可以是 要访问属性的字符串 */ 
obj2["name"]	
/* 
 * 当访问的属性是变化时, 可设置变量来代替, 如果属性名中有 
 * 如果属性名中包含会导致语法错误的字符，或者属性名使用的是关键字或保留字,则需要用[]表示法访问属性
 */
for(let key in obj2) {
	console.log(obj2[key]);	//	依次输出 "name", 12
}
